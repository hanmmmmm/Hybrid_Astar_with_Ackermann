145 void ClassPath2DSegment::extendThePath()
146 {
147  double _path_size = m_path_extended_.size();
148  double _dx = m_path_extended_.back().x - m_path_extended_.at(_path_size-2).x;
149  double _dy = m_path_extended_.back().y - m_path_extended_.at(_path_size-2).y;
150  ClassPose2D _temp(0, 0, 0);
151  double _last_x = m_path_extended_.back().x;
152  double _last_y = m_path_extended_.back().y;
153 
154  // the value 7 here is arbitrary. This decides how many new points are added.
155  for(int i=1; i<7; i++)
156  {
157  _temp.x = _last_x + _dx * i;
158  _temp.y = _last_y + _dy * i;
159  _temp.yaw = m_path_extended_.back().yaw;
160  m_path_extended_.push_back(_temp);
161  }
162 }

169 nav_msgs::Path ClassPath2DSegment::toRosPath()
170 {
171  nav_msgs::Path _ros_path;
172  for (auto pose : m_path_original_)
173  {
174  geometry_msgs::PoseStamped _temp;
175  _temp.pose.position.x = pose.x;
176  _temp.pose.position.y = pose.y;
177  _temp.pose.orientation.w = 1;
178  _ros_path.poses.push_back(_temp);
179  }
180  return _ros_path;
