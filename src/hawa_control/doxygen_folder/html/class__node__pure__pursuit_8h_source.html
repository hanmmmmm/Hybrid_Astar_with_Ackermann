202 void ClassNodePurePursuit::controllerUpdate(const ros::TimerEvent &event)
203 {
204  if( ! validateOdomMsg() )
205  {
206  ROS_WARN_STREAM("Invalid robot odom msg. Do not proceed.");
207  return;
208  }
209 
210  m_mutex_path_.lock();
211 
212  ROS_INFO_STREAM("m_counter_current_segment_ " << m_segment_manager_.getCounter());
213 
214  m_segment_manager_.update(m_robot_odom_msg_.pose.pose.position.x,
215  m_robot_odom_msg_.pose.pose.position.y,
216  quaternionToEularYaw(m_robot_odom_msg_.pose.pose.orientation));
217  if (m_segment_manager_.doesPathExist())
218  {
219  m_current_path_segment_ = m_segment_manager_.getCurrentSegment();
220  m_controller_purepursuit_.setPathSegment(m_current_path_segment_);
221  nav_msgs::Path _ros_current_path = m_segment_manager_.getCurrentSegment().toRosPath();
222  _ros_current_path.header = m_segment_manager_.getOriginalPath().header;
223  m_puber__current_path_.publish(_ros_current_path);
224  }
225 
226 
227  double _signed_speed_mps = decideSpeedMps();
228  // the values for the parameters of the controller algorithm.
229  m_controller_purepursuit_.setLookAheadCoefficient(m_look_ahead_ratio_);
230  m_controller_purepursuit_.setTargetLinearSpeedMps(_signed_speed_mps);
231  m_controller_purepursuit_.setVehicleWheelbaseMeter(0.25);
232  m_controller_purepursuit_.setRobotPose(m_robot_odom_msg_.pose.pose.position.x,
233  m_robot_odom_msg_.pose.pose.position.y,
234  quaternionToEularYaw(m_robot_odom_msg_.pose.pose.orientation));
235 
236  bool _valid_target_point = false;
237  m_controller_purepursuit_.findTargetPoint(_valid_target_point);
238  if( ! _valid_target_point )
239  {
240  ROS_WARN_STREAM("Invalid target point. Do not proceed.");
241  m_mutex_path_.unlock();
242  return;
243  }
244 
245  visulizeTargetPoint();
246 
247  bool _success; double _steer_needed = 0.0;
248  m_controller_purepursuit_.solveForSpeedCommand(_success, _steer_needed);
249 
250  std::cout << _steer_needed << std::endl;
251 
252  ackermann_msgs::AckermannDriveStamped _akm_cmd_msg;
253  _akm_cmd_msg.drive.speed = _signed_speed_mps;
254  _akm_cmd_msg.drive.steering_angle = _steer_needed;
255  _akm_cmd_msg.header.stamp = ros::Time::now();
256 
257  m_puber__ackerman_msg_.publish(_akm_cmd_msg);
258 
259  m_mutex_path_.unlock();
260 }

281 double ClassNodePurePursuit::calcSpeedRampping()
282 {
283  double _dist_to_end = m_segment_manager_.getDistToEnd();
284  double _dist_threshold = 1.0; // meter
285  double _min_speed_mps = 0.4; // meter per second, this the mininum value of the output.
286  if (_dist_to_end >= _dist_threshold)
287  {
288  return m_expected_speed_mps_;
289  }
290  double _slope = (m_expected_speed_mps_ - _min_speed_mps) / _dist_threshold;
291 
292  return _slope * _dist_to_end + _min_speed_mps;
293 }

301 double ClassNodePurePursuit::decideSpeedMps()
302 {
303  double _spd_val = 0;
304  if (m_current_path_segment_.isForward())
305  {
306  // _spd_val = std::abs(m_expected_speed_mps_);
307  _spd_val = std::abs(calcSpeedRampping());
308  }
309  else
310  {
311  // _spd_val = - std::abs(m_expected_speed_mps_);
312  _spd_val = - std::abs(calcSpeedRampping());
313  }
314 
315  if (m_segment_manager_.didFinishAll())
316  {
317  _spd_val = 0;
318  }
319  return _spd_val;
320 }
