140 std::vector<int> ClassHawaMultiSegmentManager::findSingularPoints()
141 {
142  std::vector<int> _result;
143  size_t _num_points = m_original_path_.poses.size();
144 
145  if (_num_points < 3)
146  return _result;
147 
148  std::vector<geometry_msgs::PoseStamped>& _ptr_poses = m_original_path_.poses;
149 
150  for(int i=1; i<_num_points-1; i++)
151  {
152  std::array<double, 3> _point_last = {_ptr_poses.at(i-1).pose.position.x,
153  _ptr_poses.at(i-1).pose.position.y,
154  quaternionToEularYaw(_ptr_poses.at(i-1).pose.orientation)};
155  std::array<double, 3> _point_this = {_ptr_poses.at(i).pose.position.x,
156  _ptr_poses.at(i).pose.position.y,
157  quaternionToEularYaw(_ptr_poses.at(i).pose.orientation)};
158  std::array<double, 3> _point_next = {_ptr_poses.at(i+1).pose.position.x,
159  _ptr_poses.at(i+1).pose.position.y,
160  quaternionToEularYaw(_ptr_poses.at(i+1).pose.orientation)};
161 
162  if (calcAngleByThreePoints(_point_last, _point_this, _point_next) < M_PI/2.0)
163  {
164  _result.push_back(i);
165  }
166  }
167  return _result;
168 }

175 void ClassHawaMultiSegmentManager::split_whole_path(std::vector<int> singular_points)
176 {
177  ROS_INFO_STREAM("split_whole_path(), " << singular_points.size() << " sg points.");
178 
179  m_vector_segments_.clear();
180 
181  std::vector<std::array<int,2>> _range_of_all_segments;
182 
183  if (singular_points.size() == 0)
184  {
185  _range_of_all_segments.push_back(std::array<int,2>{0, int(m_original_path_.poses.size()-1)});
186  }
187  else if (singular_points.size() > 0)
188  {
189  _range_of_all_segments.push_back(std::array<int,2>{0,
190  singular_points[0]});
191  for(int i=0; i<singular_points.size()-1; i++)
192  {
193  _range_of_all_segments.push_back(std::array<int,2>{singular_points[i],
194  singular_points[i+1]});
195  }
196  _range_of_all_segments.push_back(std::array<int,2>{singular_points.back(),
197  int(m_original_path_.poses.size()-1)});
198  }
199 
200  for (auto rg : _range_of_all_segments)
201  {
202  ClassPath2DSegment _one_segment;
203  for (int ct=rg[0]; ct<rg[1]+1; ct++)
204  {
205  _one_segment.pushbackForOriginal(ClassPose2D(m_original_path_.poses.at(ct).pose.position.x,
206  m_original_path_.poses.at(ct).pose.position.y,
207  quaternionToEularYaw( m_original_path_.poses.at(ct).pose.orientation)));
208  _one_segment.pushbackForExtended(ClassPose2D(m_original_path_.poses.at(ct).pose.position.x,
209  m_original_path_.poses.at(ct).pose.position.y,
210  quaternionToEularYaw( m_original_path_.poses.at(ct).pose.orientation)));
211  }
212  _one_segment.extendThePath();
213  if (estimateDirectionIsForward(_one_segment))
214  {
215  _one_segment.setDirectionForward();
216  }
217  else
218  {
219  _one_segment.setDirectionBackward();
220  }
221 
222  m_vector_segments_.push_back(_one_segment);
223  }
224  ROS_INFO_STREAM("num of segments: " << m_vector_segments_.size());
225 }

252 bool ClassHawaMultiSegmentManager::estimateDirectionIsForward(ClassPath2DSegment& r_segment)
253 {
254  if (r_segment.m_path_extended_.size() < 2)
255  {
256  std::cerr << "estimateDirectionIsForward. Not enough points. " << std::endl;
257  }
258 
259  ClassPose2D p1 = r_segment.m_path_extended_[0];
260  ClassPose2D p2 = r_segment.m_path_extended_[1];
261 
262  double _dx = p2.x - p1.x;
263  double _dy = p2.y - p1.y;
264  double _yaw_p2_to_p1 = std::atan2(_dy, _dx);
265  _yaw_p2_to_p1 = mod2pi(_yaw_p2_to_p1);
266  double _yaw_p1 = mod2pi(p1.yaw);
267  double _large = std::max(_yaw_p2_to_p1, _yaw_p1);
268  double _small = std::min(_yaw_p2_to_p1, _yaw_p1);
269  double _diff = _large - _small;
270  if (_diff > M_PI)
271  {
272  _diff = 2*M_PI - _diff;
273  }
274 
275  return (_diff < (M_PI/2.0));
276 }
