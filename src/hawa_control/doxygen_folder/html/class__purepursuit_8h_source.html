195 void ClassPurePursuit::findTargetPoint(bool& success)
196 {
197  int _num_points = m_the_segment_.m_path_extended_.size();
198  if(_num_points <= 2)
199  {
200  ROS_WARN_STREAM("findTargetPoint() Path too short: " << _num_points);
201  success = false;
202  return;
203  }
204 
205  computeLookAheadDistnace();
206  m_vector_target_points_.clear();
207 
208  for(int i=0; i<_num_points-1; i++)
209  {
210  ClassPose2D _pose_1 = m_the_segment_.m_path_extended_.at(i);
211  ClassPose2D _pose_2 = m_the_segment_.m_path_extended_.at(i+1);
212  double _dist_1 = computeDistanceMeter(_pose_1.x, _pose_1.y, m_robot_pose_.x_meter, m_robot_pose_.y_meter);
213  double _dist_2 = computeDistanceMeter(_pose_2.x, _pose_2.y, m_robot_pose_.x_meter, m_robot_pose_.y_meter);
214 
215  if( (_dist_1 < m_parameters_.look_ahead_distance_meter_)
216  && (_dist_2 > m_parameters_.look_ahead_distance_meter_) )
217  {
218  geometry_msgs::Point _p1;
219  _p1.x = _pose_1.x;
220  _p1.y = _pose_1.y;
221 
222  geometry_msgs::Point _p2;
223  _p2.x = _pose_2.x;
224  _p2.y = _pose_2.y;
225 
226  m_vector_target_points_.push_back(std::array<geometry_msgs::Point, 2> {_p1, _p2});
227  }
228 
229  }
230 
231  if(m_vector_target_points_.size() == 0)
232  {
233  std::cout << "No Target point found." << std::endl;
234  /*
235  TODO:
236  expand searching area and try again.
237  */

239  else if(m_vector_target_points_.size() >= 2)
240  {
241  std::cout << " >= 2 Target points found." << std::endl;
242  /*

244  select the best one.
245  */

248  {
249  std::cout << " == 1 Target point found." << std::endl;
250  success = true;

272 void ClassPurePursuit::solveForSpeedCommand(bool &r_success, double &r_steer_rad)
273 {
274  // first, compute the angle difference between the robot_heading and the line from robot
275  // to the target_point.
276  double _angle_robot_to_target_point = computeYawOf2Points(m_robot_pose_.x_meter,
277  m_robot_pose_.y_meter,
278  m_actual_target_points_.point.x,
279  m_actual_target_points_.point.y);
280  _angle_robot_to_target_point = mod2pi(_angle_robot_to_target_point);
281 
282  double _large_angle = std::max<double>(_angle_robot_to_target_point, m_robot_pose_.yaw_rad);
283  double _small_angle = std::min<double>(_angle_robot_to_target_point, m_robot_pose_.yaw_rad);
284  double _angle_difference_rad = _large_angle - _small_angle;
285  if( _angle_difference_rad > M_PI)
286  {
287  _angle_difference_rad = 2*M_PI - _angle_difference_rad;
288  }
289 
290  // the value above has only magnitude but no direction.
291  // So now, check the target_point locates on which side of the robot_heading. Due to the
292  // method I use, an extra virtual point in front of robot is required.
293  double _virtual_point_x = m_robot_pose_.x_meter + 1.0 * cos(m_robot_pose_.yaw_rad);
294  double _virtual_point_y = m_robot_pose_.y_meter + 1.0 * sin(m_robot_pose_.yaw_rad);
295 
296  bool on_left_side = checkIfPointOnLeftOfLine(m_robot_pose_.x_meter,
297  m_robot_pose_.y_meter,
298  _virtual_point_x,
299  _virtual_point_y,
300  m_actual_target_points_.point.x,
301  m_actual_target_points_.point.y);
302 
303  if( ! on_left_side)
304  {
305  _angle_difference_rad *= -1.0; // using regular meth/physics coordinate. CCW == positive.
306  }
307 
308  // the main formula to compute the requied steer angle to move to the taret_point.
309  r_steer_rad = std::atan((2 * m_parameters_.wheelbase_meter_ * std::sin(_angle_difference_rad))
310  / m_parameters_.look_ahead_distance_meter_);
311  r_success = true;
312 
313  return;
314 }

323 void ClassPurePursuit::findPreciseTargetPoint(std::array<geometry_msgs::Point, 2> two_pathpoints,
324  geometry_msgs::PointStamped &result_point)
325 {
326  geometry_msgs::Point p1 = two_pathpoints[0];
327  geometry_msgs::Point p2 = two_pathpoints[1];
328  geometry_msgs::Point pm; // the point between p1 and p2.
329 
330  double _critical_distance_meter = 0.01;
331 
332  // The method is like binary search. Get the middle point bwtn p1 p2, and check if this point is close enough
333  // to the look_ahead_distance circle. If close enough or for loop ends, then use this point as result.
334  // Otherwise, keep slicing into halves.
335  // Limit the number of iterations to be tried, in the worst case.
336  for(int ct=0; ct<10; ct++)
337  {
338  pm.x = (p1.x + p2.x) /2.0;
339  pm.y = (p1.y + p2.y) /2.0;
340 
341  double _distance_pm_to_robot = computeDistanceMeter(pm.x, pm.y, m_robot_pose_.x_meter, m_robot_pose_.y_meter);
342  if( std::abs(_distance_pm_to_robot - m_parameters_.look_ahead_distance_meter_) < _critical_distance_meter )
343  {
344  result_point.point.x = pm.x;
345  result_point.point.y = pm.y;
346  return;
347  }
348  else
349  {
350  if(_distance_pm_to_robot > m_parameters_.look_ahead_distance_meter_)
351  {
352  p2 = pm;
353  }
354  else
355  {
356  p1 = pm;
357  }
358  }
359  }
360  ROS_DEBUG_STREAM_THROTTLE(4, "Reached max TargetPoint search iterations.");
361  result_point.point.x = pm.x;
362  result_point.point.y = pm.y;
363 }
