78 void StructMotionModel::prepare_model()
79 {
80  MotionPrimitive _temp_mp;
81  double angle, curve_dx, curve_dy;
82 
83  angle = std::abs(parameters.step_length_forward / parameters.turning_radius);
84 
85  curve_dx = parameters.turning_radius * std::sin(angle);
86  curve_dy = parameters.turning_radius * (1.0 - std::cos(angle));
87 
88  // first 3 options are moving forward
89  _temp_mp.edge_cost = parameters.step_length_forward * cost_multipler.forward;
90  _temp_mp.direction = EnumForwardBackward::forward;
91 
92  // to front left
93  _temp_mp.dx = curve_dx;
94  _temp_mp.dy = curve_dy;
95  _temp_mp.dyaw = angle;
96  _temp_mp.motion_type = EnumHAMotionType::F_L;
97  motions.push_back(_temp_mp);
98 
99  // to front center
100  _temp_mp.dx = parameters.step_length_forward;
101  _temp_mp.dy = 0;
102  _temp_mp.dyaw = 0;
103  _temp_mp.motion_type = EnumHAMotionType::F_S;
104  motions.push_back(_temp_mp);
105 
106  // to front right
107  _temp_mp.dx = curve_dx;
108  _temp_mp.dy = -curve_dy;
109  _temp_mp.dyaw = -angle;
110  _temp_mp.motion_type = EnumHAMotionType::F_R;
111  motions.push_back(_temp_mp);
112 
113  angle = std::abs(parameters.step_length_backward / parameters.turning_radius);
114 
115  curve_dx = parameters.turning_radius * sin(angle);
116  curve_dy = parameters.turning_radius * (1.0 - cos(angle));
117 
118  // next 3 options are moving backward
119  _temp_mp.edge_cost = parameters.step_length_backward * cost_multipler.backward;
120  _temp_mp.direction = EnumForwardBackward::backward;
121 
122  // to back right
123  _temp_mp.dx = -curve_dx;
124  _temp_mp.dy = -curve_dy;
125  _temp_mp.dyaw = angle;
126  _temp_mp.motion_type = EnumHAMotionType::R_R;
127  motions.push_back(_temp_mp);
128 
129  // to back center
130  _temp_mp.dx = -parameters.step_length_backward;
131  _temp_mp.dy = 0;
132  _temp_mp.dyaw = 0;
133  _temp_mp.motion_type = EnumHAMotionType::R_S;
134  motions.push_back(_temp_mp);
135 
136  // to back left

138  _temp_mp.dy = curve_dy;
139  _temp_mp.dyaw = -angle;
140  _temp_mp.motion_type = EnumHAMotionType::R_L;
141  motions.push_back(_temp_mp);
