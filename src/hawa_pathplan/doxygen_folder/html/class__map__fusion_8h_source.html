207 void ClassMapFusion::mergeData( const ros::TimerEvent &event )
208 {
209  if( ! m_received_map_ ) return;
210 
211  ClassHawaTimer _timer;
212  _timer.startNow();
213 
214  m_map_mutex_.lock();
215  if (m_enable_scan_) m_scan_mutex_.lock();
216  if (m_depth_enabled_) m_depth_mutex_.lock();
217 
218  nav_msgs::OccupancyGrid _new_map_to_pub;
219  _new_map_to_pub.data = m_map_msg_.data;
220  _new_map_to_pub.header = m_map_msg_.header;
221  _new_map_to_pub.info = m_map_msg_.info;
222 
223  if (m_enable_scan_)
224  {
225  putLidarIntoGrids(_new_map_to_pub, m_scan_msg_);
226  }
227  else
228  {
229  // inflate the static section in the map
230  int _index_1D;
231  for(int x_ind = 0; x_ind<_new_map_to_pub.info.width; x_ind++)
232  {
233  for(int y_ind = 0; y_ind<_new_map_to_pub.info.height; y_ind++)
234  {
235  _index_1D = convertGrid2dTo1d(x_ind, y_ind);
236  if (m_map_msg_.data.at(_index_1D) > 70)
237  {
238  inflateOneCell(&_new_map_to_pub, &m_inflation_samples, x_ind, y_ind);
239  }
240  }
241  }
242  }
243 
244  m_puber_map_.publish( _new_map_to_pub );
245 
246  m_map_mutex_.unlock();
247  if (m_enable_scan_) m_scan_mutex_.unlock();
248  if (m_depth_enabled_) m_depth_mutex_.unlock();
249 
250  _timer.endTiming();
251 
252  ROS_DEBUG_STREAM_THROTTLE(30, "map fusion "<< int(_timer.getDuration()*1000.0) << " ms");
253 }

307 void ClassMapFusion::inflateOneCell(nav_msgs::OccupancyGrid *ptr_map_msg,
308  ClassInflationSamples *ptr_inflation,
309  const int center_x,
310  const int center_y
311  )
312 {
313  int _radius = ptr_inflation->getRadius();
314 
315  int x_start_in_map = std::max(center_x - _radius, 0);
316  int x_end_in_map = std::min(center_x + _radius + 1, int(ptr_map_msg->info.width));
317  int x_start_in_sample= std::max(_radius - center_x, 0);
318  int x_end_in_sample = std::min(_radius + int(ptr_map_msg->info.width) - 1 - center_x, _radius * 2 + 1);
319 
320  int y_start_in_map = std::max(center_y - _radius, 0);
321  int y_end_in_map = std::min(center_y + _radius + 1, int(ptr_map_msg->info.height));
322  int y_start_in_sample= std::max(_radius - center_y, 0);
323  int y_end_in_sample = std::min(_radius + int(ptr_map_msg->info.height) - 1 - center_y, _radius * 2 + 1);
324 
325  int index_1D;
326  int x_counter = 0;
327  int y_counter = 0;
328  for( int y_ind = y_start_in_map; y_ind<y_end_in_map; y_ind++ )
329  {
330  x_counter = 0;
331  for( int x_ind = x_start_in_map; x_ind<x_end_in_map; x_ind++ )
332  {
333  index_1D = convertGrid2dTo1d(x_ind, y_ind);
334  int _temp_y = y_start_in_sample + y_counter;
335  int _temp_x = x_start_in_sample + x_counter;
336  if(ptr_inflation->m_inflate_sample_.at(_temp_y).at(_temp_x) > 0)
337  {
338  ptr_map_msg->data.at(index_1D) = std::max(ptr_inflation->m_inflate_sample_.at(_temp_y).at(_temp_x),
339  ptr_map_msg->data.at(index_1D));
340  }
341  x_counter ++ ;
342  }
343  y_counter ++ ;
344  }
345 }

352 void ClassMapFusion::putLidarIntoGrids(nav_msgs::OccupancyGrid &r_grids, sensor_msgs::LaserScan &r_scan)
353 {
354  getLidarPoseInMapFrame();
355 
356  double _angle = r_scan.angle_min;
357 
358  double _angle_inc = r_scan.angle_increment;
359 
360  tf::Transform _point_tf_in_lidar, _point_tf_in_map;
361 
362  double _x, _y, _x_in_map, _y_in_map;
363  int _x_index, _y_index, _index_1D;
364  for( float range : r_scan.ranges )
365  {
366  try
367  {
368  if( range > 20.0)
369  {
370  _angle += _angle_inc ;
371  continue;
372  }
373  _x = range * std::sin( -_angle + M_PI/2.0 );
374  _y = range * std::cos( -_angle + M_PI/2.0 );
375 
376  _point_tf_in_lidar.setOrigin(tf::Vector3(_x, _y, 0));
377  _point_tf_in_map = m_lidar_to_base_tf_ * _point_tf_in_lidar;
378 
379  _x_in_map = _point_tf_in_map.getOrigin().x();
380  _y_in_map = _point_tf_in_map.getOrigin().y();
381 
382  convertRealToGrid(_x_in_map, _y_in_map, _x_index, _y_index);
383  _index_1D = convertGrid2dTo1d(_x_index, _y_index);
384  inflateOneCell(&r_grids, &m_inflation_samples, _x_index, _y_index);
385  }
386  catch (tf::TransformException ex)
387  {
388  ROS_ERROR_STREAM("getLidarPoseInMapFrame: " << ex.what());
389  ros::Duration(0.5).sleep();
390  }
391  _angle += _angle_inc ;
392  }
393 }
