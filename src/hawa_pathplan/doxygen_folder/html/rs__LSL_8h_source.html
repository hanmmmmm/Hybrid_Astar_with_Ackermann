75 StructReedsSheppFormulaResult LpSpLp_base(StructPoseReal* ptr_pose)
76 {
77  StructReedsSheppFormulaResult _rs_result;
78 
79  convertCartesianToPolar(ptr_pose->x - std::sin(ptr_pose->yaw),
80  ptr_pose->y + std::cos(ptr_pose->yaw) - 1,
81  _rs_result.u,
82  _rs_result.t);
83  _rs_result.v = mod2pi(ptr_pose->yaw - _rs_result.t);
84  _rs_result.L = std::abs(_rs_result.t) + std::abs(_rs_result.u) + std::abs(_rs_result.v) ;
85  if( _rs_result.t > 0.0 && _rs_result.v > 0.0){
86  _rs_result.valid = true;
87  }
88  else{
89  _rs_result.valid = false;
90  }
91 
92  if (std::abs(_rs_result.t) > 1.5 * M_PI || std::abs(_rs_result.v) > 1.5 * M_PI) _rs_result.valid = false;
93 
94  return _rs_result;
95 }
