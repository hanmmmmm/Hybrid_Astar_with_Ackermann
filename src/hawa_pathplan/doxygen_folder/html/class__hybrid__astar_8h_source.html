268 bool ClassHybridAStar::search()
269 {
270  ROS_INFO_STREAM("ClassHybridAStar::search() start.");
271 
272  ClassHawaTimer _timer;
273  _timer.startNow();
274 
275  int _search_iteration_count = 0;
276  while (!m_flags_.reach_goal)
277  {
278  // check if there's any valid nodes to search.
279  if (m_opennodes_pq_.size() == 0)
280  {
281  m_flags_.trapped = true;
282  ROS_WARN_STREAM("ClassHybridAStar trapped. Tried " << _search_iteration_count << " times");
283  return false; // there is no more OPEN node to search
284  }
285 
286  exploreOneNode();
287 
288  _search_iteration_count++;
289 
290  if (m_flags_.found_rs_solution)
291  {
292  ROS_INFO_STREAM("search success, found_rs_solution, tried " << _search_iteration_count << " times");
293  break;
294  }
295 
296  if (int(_timer.getDurationNonStop() * 1000.0) >= m_parameters_.time_out_ms)
297  {
298  ROS_WARN_STREAM("ClassHybridAStar search timeout, failed, tried " << _search_iteration_count << " times");
299  m_flags_.timeout = true;
300  return false;
301  }
302  }
303 
304  _timer.endTiming();
305 
306  ROS_INFO_STREAM("ClassHybridAStar::search() Found the goal." << int(_timer.getDuration() * 1000.0) << "ms. Tried " << _search_iteration_count << " iterations. RS_search "
307  << m_timer_and_counter_.count__rs_search_.getVal() << " times.");
308 
309  ROS_DEBUG_STREAM("ClassHybridAStar::search() finish.");
310 
311  return true;
312 }

320 bool ClassHybridAStar::tryFindReedsSheppCurve(const StructPoseGrid &r_curr_grid)
321 {
322  m_timer_and_counter_.count__rs_search_.increment();
323 
324  ClassHawaTimer _timer;
325  _timer.startNow();
326 
327  try
328  {
329  int _curr_grid_index_1d = m_gridmap_handler_.convert_3D_to_oneD(r_curr_grid.x, r_curr_grid.y, r_curr_grid.yaw);
330  std::array<double, 3> _rs_start_pose_array = m_gridinfo_vector_.at(_curr_grid_index_1d).real_pose.to_array3();
331  std::array<double, 3> _rs_goal_pose_array = m_important_poses_.goal_pose.to_array3();
332 
333  StructPoseReal _rs_start_pose(_rs_start_pose_array.at(0), _rs_start_pose_array.at(1), _rs_start_pose_array.at(2));
334  StructPoseReal _rs_goal_pose(_rs_goal_pose_array.at(0), _rs_goal_pose_array.at(1), _rs_goal_pose_array.at(2));
335 
336  m_RS_curve_finder_.setup(_rs_start_pose, _rs_goal_pose);
337  m_RS_curve_finder_.search();
338  }
339  catch (const std::exception &e)
340  {
341  ROS_WARN_STREAM_NAMED("ClassHybridAStar::tryFindReedsSheppCurve", "m_RS_curve_finder_ try_catch exception:" << e.what());
342  }
343 
344  _timer.endTiming();
345  m_timer_and_counter_.sec__rs_search += _timer.getDuration();
346 
347  if (m_RS_curve_finder_.m_vector_path_results_.size() == 0)
348  return false;
349 
350  _timer.startNow();
351 
352  bool _found_usable_RS_path = checkRSSearchResult();
353  if (_found_usable_RS_path)
354  {
355  m_important_poses_.grid_last_incremental_step = r_curr_grid;
356  }
357 
358  _timer.endTiming();
359  m_timer_and_counter_.sec__rs_collision_check += _timer.getDuration();
360 
361  return _found_usable_RS_path;
362 }

370 bool ClassHybridAStar::checkRSSearchResult()
371 {
372  ClassHawaTimer _timer;
373  _timer.startNow();
374 
375  bool _found_it = false;
376  for (auto one_rs_path : m_RS_curve_finder_.m_vector_path_results_)
377  {
378  if (_found_it)
379  {
380  break;
381  }
382  if ((!one_rs_path.valid) || (one_rs_path.path_steps.size() <= 0))
383  {
384  continue;
385  }
386 
387  bool _this_path_is_usable = true;
388  m_path_RS_section_.clear();
389 
390  StructPoseGrid _nb_grid;
391  StructPoseReal _nb_real;
392 
393  for (auto step : one_rs_path.path_steps)
394  {
395  if (!_this_path_is_usable)
396  break;
397 
398  try
399  {
400  _nb_real.x = step[0];
401  _nb_real.y = step[1];
402  _nb_real.yaw = mod2pi(step[2]);
403 
404  m_gridmap_handler_.convert_fine_pose_to_grid(_nb_real, _nb_grid);
405 
406  if (m_gridmap_handler_.check_grid_within_map(_nb_grid.x, _nb_grid.y))
407  {
408  bool step_is_clear = m_gridmap_handler_.check_grid_clear(_nb_grid.x,
409  _nb_grid.y,
410  ClassGridMapHandler::EnumMode::plan);
411  if (!step_is_clear)
412  {
413  _this_path_is_usable = false;
414  }
415  }
416  m_path_RS_section_.push_back(_nb_real);
417  }
418  catch (const std::exception &e)
419  {
420  ROS_WARN_STREAM_NAMED("ClassHybridAStar::checkRSSearchResult()", "" << e.what());
421  }
422  }
423  _found_it = _this_path_is_usable;
424  }
425  return _found_it;
426 }

467 void ClassHybridAStar::exploreOneNode()
468 {
469  ClassHawaTimer _timer;
470  _timer.startNow();
471 
472  bool _this_node_is_doable = false;
473  GridInfo _current_grid;
474 
475  prepareForExplore(&_this_node_is_doable, &_current_grid);
476 
477  if (!_this_node_is_doable)
478  return;
479 
480  if (m_rs_search_trigger_.check_match())
481  {
482  m_flags_.found_rs_solution = tryFindReedsSheppCurve(_current_grid.self_grid);
483  if (m_flags_.found_rs_solution)
484  return;
485  }
486  m_rs_search_trigger_.increment_counter();
487 
488  double _curr_theta = _current_grid.real_pose.yaw;
489  double _cos_theta = std::cos(_curr_theta);
490  double _sin_theta = std::sin(_curr_theta);
491 
492  for (auto mm : m_motion_model_.motions)
493  {
494  GridInfo _neigbr_grid;
495  double _step_dx = _cos_theta * mm.dx + _sin_theta * mm.dy;
496  double _step_dy = _sin_theta * mm.dx + _cos_theta * mm.dy;
497  _neigbr_grid.real_pose.x = _current_grid.real_pose.x + _step_dx;
498  _neigbr_grid.real_pose.y = _current_grid.real_pose.y + _step_dy;
499  _neigbr_grid.real_pose.yaw = mod2pi(_curr_theta + mm.dyaw);
500  // _neigbr_grid.steer_type = count;
501  _neigbr_grid.motion_type = mm.motion_type;
502 
503  m_gridmap_handler_.convert_fine_pose_to_grid(_neigbr_grid.real_pose, _neigbr_grid.self_grid);
504 
505  bool _outside_of_map = !m_gridmap_handler_.check_grid_within_map(_neigbr_grid.self_grid.x,
506  _neigbr_grid.self_grid.y);
507  bool _grid_inaccessible = !m_gridmap_handler_.check_grid_clear(_neigbr_grid.self_grid.x,
508  _neigbr_grid.self_grid.y,
509  ClassGridMapHandler::EnumMode::plan);
510  if (_outside_of_map || _grid_inaccessible)
511  {
512  continue;
513  }
514 
515  int _neighbor_grid_index_1d = m_gridmap_handler_.convert_3D_to_oneD(_neigbr_grid.self_grid.x,
516  _neigbr_grid.self_grid.y,
517  _neigbr_grid.self_grid.yaw);
518  double _new_g_cost = _current_grid.gcost + mm.edge_cost * estimate_changing_motion_type_cost(_current_grid.motion_type,
519  mm.motion_type);
520  double _new_f_cost = _new_g_cost + computeHCostEuclidean(_neigbr_grid.real_pose, m_important_poses_.goal_pose);
521 
522  bool _is_new = m_gridinfo_vector_.at(_neighbor_grid_index_1d).grid_type == AStarGridType::NewGrid;
523  bool _is_open = m_gridinfo_vector_.at(_neighbor_grid_index_1d).grid_type == AStarGridType::Open;
524  bool _better_new_parent = _new_f_cost < m_gridinfo_vector_.at(_neighbor_grid_index_1d).fcost;
525 
526  if (_is_new || (_is_open && _better_new_parent))
527  {
528  m_gridinfo_vector_.at(_neighbor_grid_index_1d).grid_type = AStarGridType::Open;
529  m_gridinfo_vector_.at(_neighbor_grid_index_1d).parent_grid = _current_grid.self_grid;
530  m_gridinfo_vector_.at(_neighbor_grid_index_1d).self_grid = _neigbr_grid.self_grid;
531  m_gridinfo_vector_.at(_neighbor_grid_index_1d).real_pose = _neigbr_grid.real_pose;
532  m_gridinfo_vector_.at(_neighbor_grid_index_1d).motion_type = _neigbr_grid.motion_type;
533  m_gridinfo_vector_.at(_neighbor_grid_index_1d).gcost = _new_g_cost;
534  m_gridinfo_vector_.at(_neighbor_grid_index_1d).fcost = _new_f_cost;
535 
536  m_opennodes_pq_.push(NodeinfoForPQ(
537  m_gridinfo_vector_.at(_neighbor_grid_index_1d).fcost,
538  m_gridinfo_vector_.at(_neighbor_grid_index_1d).self_grid));
539  }
540  }
541  _timer.endTiming();
542  m_timer_and_counter_.sec__grid_search += _timer.getDuration();
543 }

550 void ClassHybridAStar::getFinalHybridAstarPath(ClassCustomPathContainer &r_path)
551 {
552  ROS_DEBUG_STREAM("ClassHybridAStar::getFinalHybridAstarPath() start.");
553  r_path.clear_points();
554 
555  StructPoseGrid temp_step = m_important_poses_.grid_last_incremental_step;
556  if (m_flags_.trapped || m_flags_.timeout)
557  {
558  temp_step = m_important_poses_.min_cost_node;
559  }
560 
561  while (temp_step != m_important_poses_.start_grid)
562  {
563  int _temp_step_index_1d = m_gridmap_handler_.convert_3D_to_oneD(temp_step.x, temp_step.y, temp_step.yaw);
564  std::array<double, 3> _temp_step_pose = m_gridinfo_vector_[_temp_step_index_1d].real_pose.to_array3();
565 
566  r_path.pushfront(_temp_step_pose);
567  temp_step = m_gridinfo_vector_[_temp_step_index_1d].parent_grid;
568 
569  if (temp_step.x == 0 && temp_step.y == 0)
570  {
571  break;
572  }
573  }
574  int _temp_step_index_1d = m_gridmap_handler_.convert_3D_to_oneD(temp_step.x, temp_step.y, temp_step.yaw);
575  r_path.pushfront(m_gridinfo_vector_[_temp_step_index_1d].real_pose.to_array3()); // this one should be the starting point.
576 
577  if (!(m_flags_.trapped || m_flags_.timeout))
578  {
579  for (auto point : m_path_RS_section_)
580  {
581  r_path.pushback(point.to_array3());
582  }
583  }
584 
585  ROS_DEBUG_STREAM("ClassHybridAStar::getFinalHybridAstarPath() Done.");
586 }
