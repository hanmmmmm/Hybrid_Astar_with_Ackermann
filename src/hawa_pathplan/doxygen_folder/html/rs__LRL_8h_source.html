43 StructDubinsCurveCCCvalcollection prepareCCCLRL(StructPoseReal goal_pose,
44  StructPoseReal start_pose,
45  double turning_radius)
46 {
47  double _dyy = goal_pose.y - start_pose.y;
48  double _dxx = goal_pose.x - start_pose.x;
49  double _theta = std::atan2(_dyy, _dxx);
50  double _D = std::sqrt( _dxx*_dxx + _dyy*_dyy );
51  double _ccc_d_ = _D/turning_radius;
52  double _ccc_alpha_ = mod2pi( start_pose.yaw - _theta );
53  double _ccc_beta_ = mod2pi( goal_pose.yaw - _theta );
54 
55  double u = mod2pi( 2*M_PI - std::acos(( 6-_ccc_d_*_ccc_d_ +2*std::cos(_ccc_alpha_-_ccc_beta_)
56  + 2*_ccc_d_*(-std::sin(_ccc_alpha_) + std::sin(_ccc_beta_)) )/8.0 ) ) ;
57  double t = mod2pi(-_ccc_alpha_ - std::atan( ( std::cos(_ccc_alpha_) - std::cos(_ccc_beta_))
58  / (_ccc_d_+std::sin(_ccc_alpha_) - std::sin(_ccc_beta_))) + u/2.0 ) ;
59  double v = mod2pi(_ccc_beta_) - _ccc_alpha_ -t + mod2pi(u);
60 
61 
62 
63  StructDubinsCurveCCCvalcollection _result;
64  _result.t = t;
65  _result.u = u;
66  _result.v = v;
67 
68  _result.valid = (t>=0.0 && u>=0.0 && v>=0.0);
69 
70  return _result;
71 }

76 void solveLpRpLp(StructPoseReal goal_pose,
77  StructPoseReal start_pose,
78  StructDubinsCurveCCCvalcollection* ptr_dubins_para,
79  StructRSPathResult* ptr_path,
80  StructSamplingProperties* ptr_sample )
81 {
82  ptr_path->setValid(ptr_dubins_para->valid);
83  if (! ptr_dubins_para->valid) return;
84  double new_t = ptr_dubins_para->t;
85  double new_u = ptr_dubins_para->u;
86  double new_v = ptr_dubins_para->v;
87  ptr_path->path_length_unitless = std::abs(new_t) + std::abs(new_u) + std::abs(new_v) ;
88 
89  double _limit = 1.5 * M_PI;
90  if (std::abs(new_t) > _limit || std::abs(new_u) > _limit || std::abs(new_v) > _limit)
91  {
92  ptr_dubins_para->valid = false;
93  return;
94  }
95  sampleOnLeftCurve(new_t, start_pose.yaw, start_pose.x, start_pose.y, ptr_path, ptr_sample);
96  sampleOnRightCurve(new_u, start_pose.yaw, start_pose.x, start_pose.y, ptr_path, ptr_sample);
97  sampleOnLeftCurve(new_v, start_pose.yaw, start_pose.x, start_pose.y, ptr_path, ptr_sample);
98 }
