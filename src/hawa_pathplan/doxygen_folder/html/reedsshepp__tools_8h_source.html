120 void tauOmega(const double u, const double v, const double xi, const double eta, const double phi,
121  double &tau, double &omega)
122 {
123  double _delta = mod2pi(u - v);
124  double _A = std::sin(u) - std::sin(_delta);
125  double _B = std::cos(u) - std::cos(_delta) - 1.0;
126  double _t1 = std::atan2(eta * _A - xi * _B, xi * _A + eta * _B);
127  double _t2 = 2.0 * (std::cos(_delta) - std::cos(v) - std::cos(u)) + 3;
128  if (_t2 < 0)
129  {
130  tau = mod2pi(_t1 + M_PI);
131  }
132  else
133  {
134  tau = mod2pi(_t1);
135  }
136  omega = mod2pi(tau - u + v - phi);
137 }

147 void sampleOnLeftCurve( const double target_angle_change,
148  double& robot_yaw, double& robotx, double& roboty,
149  StructRSPathResult* ptr_path_result,
150  StructSamplingProperties* r_properties)
151 {
152  double init_yaw = robot_yaw;
153  double init_x = robotx;
154  double init_y = roboty;
155  double theta_change = 0;
156  // double speed = 0.0;
157  double ang_step_size_local = 0.0;
158  if( target_angle_change >= 0.0 ){
159  // speed = r_properties->linear_step_size;
160  ang_step_size_local = r_properties->angular_step_size;
161  }
162  else {
163  // speed = -1.0 * r_properties->linear_step_size;
164  ang_step_size_local = -1.0 * r_properties->angular_step_size;
165  }
166 
167  while ( std::abs( theta_change ) < std::abs( target_angle_change ) ){
168  if ( std::abs( target_angle_change ) - std::abs( theta_change ) < std::abs( ang_step_size_local ) )
169  {
170  ang_step_size_local = std::abs( target_angle_change ) - std::abs( theta_change );
171  if( target_angle_change < 0.0 )
172  {
173  ang_step_size_local *= -1.0;
174  }
175  }
176 
177  double _body_frame_dx = r_properties->turning_radius * std::sin(ang_step_size_local);
178  double _body_frame_dy = r_properties->turning_radius
179  - r_properties->turning_radius * std::cos(ang_step_size_local);
180 
181  double _fix_frame_dx = _body_frame_dx * cos(robot_yaw) - _body_frame_dy * cos(M_PI/2.0 - robot_yaw);
182  double _fix_frame_dy = _body_frame_dx * cos(M_PI/2.0 - robot_yaw) + _body_frame_dy * cos(robot_yaw);
183 
184  robotx += _fix_frame_dx;
185  roboty += _fix_frame_dy;
186  theta_change += ang_step_size_local;
187  robot_yaw += ang_step_size_local ;
188 
189  // std::array<double,3> pose = {robotx, roboty, robot_yaw};
190  // ptr_path_result->path_steps.push_back(pose);
191  ptr_path_result->path_steps.push_back(std::array<double,3>{robotx, roboty, robot_yaw});
192  }
193 
194  double _body_frame_dx = r_properties->turning_radius * sin(target_angle_change);
195  double _body_frame_dy = r_properties->turning_radius - r_properties->turning_radius * cos(target_angle_change);
196  double _fix_frame_dx = _body_frame_dx * cos(init_yaw) - _body_frame_dy * cos(M_PI/2.0 - init_yaw);
197  double _fix_frame_dy = _body_frame_dx * cos(M_PI/2.0 - init_yaw) + _body_frame_dy * cos(init_yaw);
198 
199  robotx = init_x + _fix_frame_dx;
200  roboty = init_y + _fix_frame_dy;
201  robot_yaw = mod2pi(init_yaw + target_angle_change);
202 }

270 void sampleOnStraightLine(const double target_angle_change,
271  double robot_yaw, double& robotx, double& roboty,
272  StructRSPathResult* ptr_path_result,
273  StructSamplingProperties* r_properties)

279  if( target_angle_change >= 0.0 ){
280  linear_step_size_local = r_properties->linear_step_size;
281  }
282  else {
283  linear_step_size_local = -1.0 * r_properties->linear_step_size;
284  }
285 
286  double _target_distance = target_angle_change * r_properties->turning_radius;
287  bool _finish = false;
288 
289  while ( std::abs( theta_change ) < std::abs( _target_distance ) )
290  {
291  if ( std::abs( _target_distance ) - std::abs( theta_change ) < std::abs( linear_step_size_local ) )
292  {
293  linear_step_size_local = std::abs( _target_distance ) - std::abs( theta_change );
294  if( target_angle_change < 0.0 )
295  {
296  linear_step_size_local *= -1.0;
297  }
298  _finish = true;
299  }
300 
301  double dx = linear_step_size_local * cos( robot_yaw );
302  double dy = linear_step_size_local * sin( robot_yaw );
303  robotx = robotx + dx;
304  roboty = roboty + dy;
305  theta_change += linear_step_size_local;
306 
307  // std::array<double,3> pose = {robotx, roboty, robot_yaw};
308  // result_container.path_steps.push_back(pose);
309  ptr_path_result->path_steps.push_back(std::array<double,3>{robotx, roboty, robot_yaw});
310 
311  if (_finish)
312  break;
313 
314  }
315  robotx = init_x + target_angle_change * r_properties->turning_radius * std::cos( robot_yaw );
316  roboty = init_y + target_angle_change * r_properties->turning_radius * std::sin( robot_yaw );
317 }
