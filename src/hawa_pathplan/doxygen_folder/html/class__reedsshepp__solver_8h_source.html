56 /*
57 This is my implementation of Reeds-Sheep curve.
58 Most parts are following the paper.
59 One exceptation is that L+R+L+ L-R-L- R+L+R+ R-L-R- are also included;
60 so here are 52=48+4 types of curves are being considered;
61 And the CCC types are computed using Dubins Curve formula.
62 
63 The result is a vector of my custom path type, defined in another header file.
64 Each 'path' contains the type-word, unitless length, sampled waypoints (x,y,yaw), valid (bool).
65 Sorted in length increasing order.
66 */

101 void ClassReedsSheppSolver::setup( StructPoseReal start_pose, StructPoseReal goal_pose )
102 {
103  m_start_pose_ = start_pose;
104  m_goal_pose_ = goal_pose;
105 
106  m_sampling_properites_.angular_step_size = 0.2;
107  m_sampling_properites_.turning_radius = 0.5;
108  m_sampling_properites_.linear_step_size = 0.1;
109 
110  double _dy = m_goal_pose_.y - m_start_pose_.y;
111  double _dx = m_goal_pose_.x - m_start_pose_.x;
112  double _cos_theta1 = std::cos(m_start_pose_.yaw);
113  double _sin_theta1 = std::sin(m_start_pose_.yaw);
114 
115  m_goal_pose_processed_.x = ( _dx * _cos_theta1 + _dy * _sin_theta1) / m_sampling_properites_.turning_radius;
116  m_goal_pose_processed_.y = (-_dx * _sin_theta1 + _dy * _cos_theta1) / m_sampling_properites_.turning_radius;
117  m_goal_pose_processed_.yaw = mod2pi( m_goal_pose_.yaw - m_start_pose_.yaw );
118 
119  m_all_curves_.resetAllPaths();
120  while ( m_vector_path_results_.size() > 0 ){
121  m_vector_path_results_.pop_back();
122  }
123 }
