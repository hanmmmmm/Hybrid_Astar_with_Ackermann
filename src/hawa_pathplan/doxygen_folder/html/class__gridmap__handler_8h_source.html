330 bool ClassGridMapHandler::check_line_clear__by_grid(int x1, int y1, int x2, int y2, EnumMode m, EnumCollosionCheckType co)
331 {
332  if (co == EnumCollosionCheckType::end_points_only)
333  {
334  return check_grid_clear(x1, y1, m) && check_grid_clear(x2, y2, m);
335  }
336  else if (co == EnumCollosionCheckType::all_grids)
337  {
338  // use bresenham to find the grids in the middle, and check each of them.
339  return check_grid_clear(x1, y1, m) && check_grid_clear(x2, y2, m);
340  }
341  else if (co == EnumCollosionCheckType::spaced)
342  {
343  // similar to above, but check only some grids.
344  return check_grid_clear(x1, y1, m) && check_grid_clear(x2, y2, m);
345  }
346  else
347  {
348  return check_grid_clear(x1, y1, m) && check_grid_clear(x2, y2, m);
349  }
350 }

377 bool ClassGridMapHandler::convert_fine_pose_to_grid(double fine_x, double fine_y, double fine_yaw, int &grid_x, int &grid_y, int &grid_yaw)
378 {
379  if (K_ratio_fine_to_grid_xy_ > 0.001 && K_ratio_fine_to_grid_yaw_ > 0.001)
380  {
381  grid_x = int(fine_x / K_ratio_fine_to_grid_xy_);
382  grid_y = int(fine_y / K_ratio_fine_to_grid_xy_);
383  grid_yaw = int(fine_yaw / K_ratio_fine_to_grid_yaw_);
384  return true;
385  }
386  else
387  {
388  cout << "K_ratio_fine_to_grid_ seems too small to be reasonable: " << K_ratio_fine_to_grid_xy_ << " , " << K_ratio_fine_to_grid_yaw_ << endl;
389  return false;
390  }
391 }
