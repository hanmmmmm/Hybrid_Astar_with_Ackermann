255 double estimate_steering_cost(const int last_steer, const int new_steer)
256 {
257  int _diff = std::abs(last_steer - new_steer);
258  if (_diff == 0)
259  {
260  return 1.0;
261  }
262  else if (_diff == 1)
263  {
264  return 1.3;
265  }
266  else if (_diff > 0)
267  {
268  return 2.0;
269  }
270  else
271  {
272  return 1.0;
273  }
274 }

362 inline double computeHCostEuclidean(const StructPoseReal n, const StructPoseReal g)
363 {
364  double dx = n.x - g.x;
